Step 6: OpenSCAD Manager¶

Most of the analysis of a design will be done using an STL file which describes a part we create. Before we worry about generating the parts, we can set up a Python class that will manage access to OpenSCAD from the command line.

Feature 2: Add OpenSCAD Management class¶

The primary purpose of the OpenSCAD manager class is to isolate running OpenSCAD to perform needed operations. For our first test of this class, we will launch OpenSCAD and fetch the version of the program we are using. Unfortunately, getting the Continuous Integration servers to install the version I have on my machine (installed with Homebrew) has proven difficult, so we will not test the actual version, just ensure we get valid output.

The code we need to write will assume that OpenSCAD is on the system path and can be launched with the name openscad.

Users will need to install at least OpenSCAD version 2019.05, so we will set up a test to check that.

Mass Property Generator¶

Now that we can generate an STL file, we can analyze that file to get the part properties. These will be saved in a JSON output file:

mmdesigner/OpenSCAD.py¶
  1import os
  2import stl
  3from stl import mesh
  4import subprocess
  5
  6
  7class OpenSCAD(object):
  8    """Management class for OpenSCAD command line interface"""
  9    def __init__(self):
 10        pass
 11
 12    def get_version(self):
 13        """return installed OpenScAD version"""
 14        result = subprocess.run(['openscad','--version'], capture_output=True)
 15        version = result.stderr.decode().split()[2]
 16        return version
 17
 18    def gen_stl(self, scad_path):
 19        """ Generate STL file from specified SCAD file"""
 20        err_code = 1
 21        if not os.path.isfile(scad_path):
 22            return err_code
 23        if scad_path.endswith(".scad"):
 24            base, ext = scad_path.split(".")
 25            stl_path = base  + ".stl"
 26            cmd = [
 27                'openscad',
 28                scad_path,
 29                "-o",
 30                stl_path
 31            ]
 32
 33            result = subprocess.run(cmd, capture_output=True)
 34            err_code = result.returncode
 35            if not os.path.isfile(stl_path) or err_code != 0: # pragma: no cover
 36                err_code = 1
 37        return err_code
 38
 39    def _get_bounds(self):
 40        """set internal bounds variables"""
 41        minx = maxx = miny = maxy = minz = maxz = None
 42        obj = self.mesh
 43        for p in obj.points:
 44            # p contains (x, y, z)
 45            if minx is None:
 46                minx = p[stl.Dimension.X]
 47                maxx = p[stl.Dimension.X]
 48                miny = p[stl.Dimension.Y]
 49                maxy = p[stl.Dimension.Y]
 50                minz = p[stl.Dimension.Z]
 51                maxz = p[stl.Dimension.Z]
 52            else:
 53                maxx = max(p[stl.Dimension.X], maxx)
 54                minx = min(p[stl.Dimension.X], minx)
 55                maxy = max(p[stl.Dimension.Y], maxy)
 56                miny = min(p[stl.Dimension.Y], miny)
 57                maxz = max(p[stl.Dimension.Z], maxz)
 58                minz = min(p[stl.Dimension.Z], minz)
 59        self.minx = minx
 60        self.maxx = maxx
 61        self.miny = miny
 62        self.maxy = maxy
 63        self.minz = minz
 64        self.maxz = maxz
 65
 66    def dump_to_json(self, json_path):
 67        """write mas properties data to specified JSON file"""
 68        json = "{\n"
 69        json += '  "maxx": "%s",\n' % self.maxx
 70        json += '  "minx": "%s",\n' % self.minx
 71        json += '  "maxy": "%s",\n' % self.maxy
 72        json += '  "miny": "%s",\n' % self.miny
 73        json += '  "maxz": "%s",\n' % self.maxz
 74        json += '  "minz": "%s",\n' % self.minz
 75        json += '  "volume": "%s",\n' % self.volume
 76        json += '  "cgx": "%s",\n' % self.cgx
 77        json += '  "cgy": "%s",\n' % self.cgy
 78        json += '  "cgz": "%s",\n' % self.cgz
 79        json += '  "ixx": "%s",\n' % self.ixx
 80        json += '  "ixy": "%s",\n' % self.ixy
 81        json += '  "ixz": "%s",\n' % self.ixz
 82        json += '  "iyx": "%s",\n' % self.iyx
 83        json += '  "iyy": "%s",\n' % self.iyy
 84        json += '  "iyz": "%s",\n' % self.iyz
 85        json += '  "izx": "%s",\n' % self.izx
 86        json += '  "izy": "%s",\n' % self.izy
 87        json += '  "izz": "%s"\n' % self.izz
 88        json += '}\n'
 89
 90        with open(json_path,'w') as fout:
 91            fout.write(json)
 92
 93
 94    def get_properties(self, scad_file):
 95        """Calculate mass properties from specified SCAD file"""
 96        scad_path = os.path.abspath(scad_file)
 97        err_code = self.gen_stl(scad_path)
 98        base, ext = scad_path.split(".")
 99        stl_file = base + ".stl"
100        self.json_file = base + ".json"
101        self.mesh = mesh.Mesh.from_file(stl_file)
102        self._get_bounds()
103        volume, cog, inertia = self.mesh.get_mass_properties()
104        self.cgx, self.cgy, self.cgz = cog
105        self.volume = volume
106        self.ixx, self.ixy, self.ixz = inertia[0]
107        self.iyx, self.iyy, self.iyz = inertia[1]
108        self.izx, self.izy, self.izz = inertia[2]
109        return err_code
110
111    def get_bounds(self):
112        """Return bounds data as a list"""
113        return [
114                self.maxx, self.minx,
115                self.maxy, self.miny,
116                self.maxz, self.minz
117        ]
